      ```typescript
      useEffect(() => {
        if (!isDemo) {
          useLearnerStore.getState().setLastVisited(instrumentSlug, sessionSlug, sessionNumber);
        }
      }, [instrumentSlug, sessionSlug, sessionNumber, isDemo]);
      ```
